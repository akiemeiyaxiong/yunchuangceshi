
```bash

    "max-size": "10m",
    "max-file": "3"
  }
}
EOF

# 重启 Docker
sudo systemctl daemon-reload
sudo systemctl restart docker
```

## 四、项目部署

### 4.1 创建应用目录

```bash
# 创建项目目录
mkdir -p ~/app
cd ~/app
```

### 4.2 上传项目文件

#### 方式一：使用 scp 命令
```bash
# 从本地上传到服务器（在本地执行）
scp -r ./your-project ubuntu@43.166.220.37:~/app/

# 或使用密钥
scp -i /path/to/key.pem -r ./your-project ubuntu@43.166.220.37:~/app/
```

#### 方式二：使用 rsync
```bash
rsync -avz --progress ./your-project ubuntu@43.166.220.37:~/app/
```

#### 方式三：使用 Git 克隆
```bash
cd ~/app
git clone https://github.com/yourusername/your-project.git
```

### 4.3 Docker 部署示例

#### 示例 1：部署 Nginx 静态网站

```bash
cd ~/app

# 创建 docker-compose.yml
cat > docker-compose.yml <<EOF
version: '3.8'

services:
  nginx:
    image: nginx:alpine
    container_name: web
    ports:
      - "80:80"
    volumes:
      - ./html:/usr/share/nginx/html
    restart: unless-stopped
EOF

# 创建静态文件目录
mkdir -p html
echo "<h1>Hello from Tencent Cloud!</h1>" > html/index.html

# 启动服务
docker compose up -d
```

#### 示例 2：部署 Node.js 应用

```bash
cd ~/app

# 创建 docker-compose.yml
cat > docker-compose.yml <<EOF
version: '3.8'

services:
  app:
    build: .
    container_name: node-app
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
    restart: unless-stopped
    
  nginx:
    image: nginx:alpine
    container_name: nginx
    ports:
      - "80:80"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
    depends_on:
      - app
    restart: unless-stopped
EOF

# 创建 Dockerfile
cat > Dockerfile <<EOF
FROM node:18-alpine
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
EXPOSE 3000
CMD ["node", "server.js"]
EOF

# 构建并启动
docker compose up -d --build
```

### 5.1 配置 DNS 解析

- 主机记录: `@` 或 `www`
- 记录值: `43.166.220.37`

### 5.2 使用 Nginx Proxy Manager（推荐）

```bash
cd ~/app

mkdir -p npm
cd npm

cat > docker-compose.yml <<EOF
version: '3.8'

services:
  npm:
    image: jc21/nginx-proxy-manager:latest
    container_name: npm
    ports:
      - "80:80"
      - "81:81"
      - "443:443"
    volumes:
      - ./data:/data
      - ./letsencrypt:/etc/letsencrypt
    restart: unless-stopped
EOF

docker compose up -d
```

## 六、服务器监控与维护

### 6.1 监控资源使用

```bash
# 查看系统资源
top
htop

# 查看磁盘空间
df -h

# 查看内存使用
free -h
```

### 6.2 日志管理

```bash
# 查看系统日志
sudo journalctl -xe

# 查看 Docker 日志
sudo journalctl -u docker.service

# 清理 Docker 日志
docker system prune -a
```

### 6.3 自动备份脚本

```bash
# 创建备份目录
mkdir -p ~/backup

# 创建备份脚本
cat > ~/backup/backup.sh <<'EOF'
#!/bin/bash
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_DIR="~/backup/$DATE"
mkdir -p $BACKUP_DIR

# 备份应用数据
cp -r ~/app $BACKUP_DIR/

# 备份 Docker 卷
docker run --rm -v ~/app:/data -v $BACKUP_DIR:/backup alpine tar czf /backup/app.tar.gz -C /data .

# 保留最近 7 天的备份
cd ~/backup && ls -t | tail -n +8 | xargs rm -rf
EOF

chmod +x ~/backup/backup.sh

# 添加到定时任务（每天凌晨 2 点执行）
echo "0 2 * * * ~/backup/backup.sh" | crontab -
```

## 七、常见问题

### 7.1 连接问题

```bash

ping 43.166.220.37

# 检查端口开放情况
telnet 43.166.220.37 22
telnet 43.166.220.37 80
```

### 7.2 内存不足（2GB 内存优化）

```bash
# 添加 Swap 空间
sudo fallocate -l 2G /swapfile
sudo chmod 600 /swapfile
sudo mkswap /swapfile
sudo swapon /swapfile

# 永久启用
echo '/swapfile none swap sw 0 0' | sudo tee -a /etc/fstab

# 查看 Swap
free -h
```

### 7.3 磁盘空间不足

```bash
# 查看大文件
sudo du -h / | grep '[0-9]G\>'

# 清理 Docker
docker system prune -a --volumes

# 清理日志
sudo journalctl --vacuum-time=3d
```

## 八、安全建议

### 8.1 SSH 安全配置

```bash
# 编辑 SSH 配置
sudo nano /etc/ssh/sshd_config

# 建议修改以下配置
PermitRootLogin no
PasswordAuthentication no  # 如果使用密钥登录
MaxAuthTries 3
ClientAliveInterval 300
ClientAliveCountMax 2

# 重启 SSH
sudo systemctl restart sshd
```

### 8.2 定期更新

```bash
# 设置自动安全更新
sudo apt-get install unattended-upgrades
sudo dpkg-reconfigure unattended-upgrades
```
